772  double tmp;
773  int i;
774 
775  // first compute squared magnitudes of quaternion components - at least one
776  // will be greater than 0 since quaternion is unit magnitude
777 
778  qs2 = 0.25 * (R.xx + R.yy + R.zz + 1);
779  qx2 = qs2 - 0.5 * (R.yy + R.zz);
780  qy2 = qs2 - 0.5 * (R.zz + R.xx);

782

785  i = (qs2 > qx2 ) ?
786  ((qs2 > qy2) ? ((qs2 > qz2) ? 0 : 3) : ((qy2 > qz2) ? 2 : 3)) :
787  ((qx2 > qy2) ? ((qx2 > qz2) ? 1 : 3) : ((qy2 > qz2) ? 2 : 3));
788 
789  // compute signed quaternion components using numerically stable method
790  switch(i) {
791  case 0:
792  s_ = sqrt(qs2);
793  tmp = 0.25 / s_;
794  x_ = (R.zy - R.yz) * tmp;
795  y_ = (R.xz - R.zx) * tmp;
796  z_ = (R.yx - R.xy) * tmp;

798  case 1:

802  y_ = (R.xy + R.yx) * tmp;

816  x_ = (R.zx + R.xz) * tmp;
