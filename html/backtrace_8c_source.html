186 static void
187 lookup_section(bfd *abfd, asection *sec, void *opaque_data)
188 {
189  struct find_info *data = opaque_data;
190 
191  if (data->func)
192  return;
193 
194  if (!(bfd_get_section_flags(abfd, sec) & SEC_ALLOC))
195  return;
196 
197  bfd_vma vma = bfd_get_section_vma(abfd, sec);
198  if (data->counter < vma || vma + bfd_get_section_size(sec) <= data->counter)
199  return;
200 
201  bfd_find_nearest_line(abfd, sec, data->symbol, data->counter - vma, &(data->file), &(data->func), &(data->line));
202 }

227 static int
228 init_bfd_ctx(struct bfd_ctx *bc, const char * procname, struct output_buffer *ob)
229 {
230  bc->handle = NULL;
231  bc->symbol = NULL;
232 
233  bfd *b = bfd_openr(procname, 0);
234  if (!b) {
235  output_print(ob,"Failed to open bfd from (%s)\n" , procname);
236  return 1;
237  }
238 
239  int r1 = bfd_check_format(b, bfd_object);
240  int r2 = bfd_check_format_matches(b, bfd_object, NULL);
241  int r3 = bfd_get_file_flags(b) & HAS_SYMS;
242 
243  if (!(r1 && r2 && r3)) {
244  bfd_close(b);
245  output_print(ob,"Failed to init bfd from (%s)\n", procname);
246  return 1;
247  }
248 
249  void *symbol_table;
250 
251  unsigned dummy = 0;
252  if (bfd_read_minisymbols(b, FALSE, &symbol_table, &dummy) == 0) {
253  if (bfd_read_minisymbols(b, TRUE, &symbol_table, &dummy) < 0) {
254  free(symbol_table);
255  bfd_close(b);
256  output_print(ob,"Failed to read symbols from (%s)\n", procname);
257  return 1;
258  }
259  }
260 
261  bc->handle = b;
262  bc->symbol = symbol_table;
263 
264  return 0;
265 }

313 static void
314 _backtrace(struct output_buffer *ob, struct bfd_set *set, int depth , LPCONTEXT context)
315 {
316  char procname[MAX_PATH];
317  GetModuleFileNameA(NULL, procname, sizeof procname);
318 
319  struct bfd_ctx *bc = NULL;
320 
321  STACKFRAME frame;
322  memset(&frame,0,sizeof(frame));
323 
324  frame.AddrPC.Offset = context->Eip;
325  frame.AddrPC.Mode = AddrModeFlat;
326  frame.AddrStack.Offset = context->Esp;
327  frame.AddrStack.Mode = AddrModeFlat;
328  frame.AddrFrame.Offset = context->Ebp;
329  frame.AddrFrame.Mode = AddrModeFlat;
330 
331  HANDLE process = GetCurrentProcess();
332  HANDLE thread = GetCurrentThread();
333 
334  char symbol_buffer[sizeof(IMAGEHLP_SYMBOL) + 255];
335  char module_name_raw[MAX_PATH];
336 
337  while(StackWalk(IMAGE_FILE_MACHINE_I386,
338  process,
339  thread,
340  &frame,
341  context,
342  0,
343  SymFunctionTableAccess,
344  SymGetModuleBase, 0)) {
345 
346  --depth;
347  if (depth < 0)
348  break;
349 
350  IMAGEHLP_SYMBOL *symbol = (IMAGEHLP_SYMBOL *)symbol_buffer;
351  symbol->SizeOfStruct = (sizeof *symbol) + 255;
352  symbol->MaxNameLength = 254;
353 
354  DWORD module_base = SymGetModuleBase(process, frame.AddrPC.Offset);
355 
356  const char * module_name = "[unknown module]";
357  if (module_base &&
358  GetModuleFileNameA((HINSTANCE)module_base, module_name_raw, MAX_PATH)) {
359  module_name = module_name_raw;
360  bc = get_bc(ob, set, module_name);
361  }
362 
363  const char * file = NULL;
364  const char * func = NULL;
365  unsigned line = 0;
366 
367  if (bc) {
368  find(bc,frame.AddrPC.Offset,&file,&func,&line);
369  }
370 
371  if (file == NULL) {
372  DWORD dummy = 0;
373  if (SymGetSymFromAddr(process, frame.AddrPC.Offset, &dummy, symbol)) {
374  file = symbol->Name;
375  }
376  else {
377  file = "[unknown file]";
378  }
379  }
380  if (func == NULL) {
381  output_print(ob,"0x%x : %s : %s \n",
382  frame.AddrPC.Offset,
383  module_name,
384  file);
385  }
386  else {
387  output_print(ob,"0x%x : %s : %s (%d) : in function (%s) \n",
388  frame.AddrPC.Offset,
389  module_name,
390  file,
391  line,
392  func);
393  }
394  }

400 static LONG WINAPI
401 exception_filter(LPEXCEPTION_POINTERS info)
402 {
