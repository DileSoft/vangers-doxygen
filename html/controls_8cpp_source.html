272  SDL_GameController *ctrl = get_gamecontroller();
273 
274  if(!iControlsObj) return 0;

276  code = iControlsObj->getCode(id, i);
277  //std::cout<<"iKeyPressed code:"<<(int)code<<" id:"<<id<<" i:"<<i<<std::endl;
278  if(code) {
279  if (code & SDLK_JOYSTICK_BUTTON_MASK && joy) {
280  state = SDL_JoystickGetButton(joy, code ^ SDLK_JOYSTICK_BUTTON_MASK);
281  } else if (code & SDLK_GAMECONTROLLER_BUTTON_MASK && ctrl) {
282  state = SDL_GameControllerGetButton(ctrl, (SDL_GameControllerButton)(code ^ SDLK_GAMECONTROLLER_BUTTON_MASK));
283  } else if (code & SDLK_JOYSTICK_HAT_MASK && joy) {
284  state = SDL_JoystickGetHat(joy, (code ^ SDLK_JOYSTICK_HAT_MASK) / 10 ) == (code ^ SDLK_JOYSTICK_HAT_MASK) % 10;
285  } else if (code & SDLK_SCANCODE_MASK) {
286  state = SDL_GetKeyboardState(NULL)[SDL_GetScancodeFromKey(code)];
287  } else {
288  state = SDL_GetKeyboardState(NULL)[code];
289  }
290  if (state) {
291  return state;
292  }
293  }
294  }
295  return 0;
